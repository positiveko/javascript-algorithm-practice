# 부분집합 구하기


자연수 N이 주어지면 1부터 N까지의 원소를 갖는 집합의 부분집합을 모두 출력하는 프로그램을 작성하세요.


```js
/**
 * @param 자연수 N(1<=N<=10)
 * @example
 * input: 1
 * output:
 * 123
 * 12
 * 13
 * 1
 * 23
 * 2
 * 3
 */
function solution(n) {
  const answer = [];
  const check = new Array(n + 1).fill(0)
  
  const dfs = (num) => {
    if (num === n + 1) {
      let temp = '';
      for (let i = 1; i <= n; i++) {
        if (check[i] === 1) temp += i;
      }
      temp && answer.push(temp);
      return;
    };
    
    check[num] = 1;
    dfs(num + 1);
    check[num] = 0;
    dfs(num + 1);
  }
  
  dfs(1);
  return answer;
}

solution(3);
```
