# 순열 구하기

10이하의 N개의 자연수가 주어지면 이 중 M개를 뽑아 일렬로 나열하는 방법을 모두 출력합니다.


```js
function solution(m, arr){         
  let answer = [];
  let check = new Array(arr.length).fill(0);
  let temp = new Array(m).fill(0);
  
  const dfs = (L) => {
    if (L === m) {
      answer.push(temp.slice());
      return;
    }
    
    for (let i = 0; i < arr.length; i++) {
      if (check[i] === 0) {
        check[i] = 1;
        temp[L] = arr[i];
        dfs(L + 1);
        check[i] = 0;
      }
    }
  }
  dfs(0);
  
  return answer;
}

solution(2, [3, 6, 9]);
```
